 12class ALXhtml:
 13    def __init__(self, title: str = None) -> None:
 14        """
 15        An HTML module to simplify the creation of HTML through the use of
 16        methods.  This makes the code calling the library much more readable
 17
 18        :param title: The title to use in the head section of the HTML
 19        """
 20        self.config = ALXapp.read_lib_config(self)
 21        """Adds the configuration read from `alx.ini`"""
 22        self.logger = ALXapp.logger
 23        """The default logger from the alx.app.ALXapp.logger"""
 24        self.css = "\n" + self.config.get('html', 'css') + "\n"
 25        """Stores the css read from the ini file.  More styles can be added with `set_css`"""
 26        self.head = "<head>\n"
 27        """Initialise the `head` with `title` if specified"""
 28        if title:
 29            self.head += "<title>" + title + "</title>\n"
 30        self.doctype = "<!DOCTYPE html>\n"
 31        """Initialise the DOCTYPE"""
 32        self.html = "<html>\n"
 33        """Initialise the `html` text"""
 34        self.body = "<body>\n"
 35        """Initialise the `body` text"""
 36        self._current_row = 0
 37        """Internal variable to hold current row number"""
 38        self._current_column = 0
 39        """Internal variable to hold current column number"""
 40        self.column_headings = []
 41        """Holds column indexes that should be treated as headings.
 42        For example, to mark the first column as a heading, set to 
 43        `[True]` through `set_column_headings`"""
 44        self.column_alignments = []
 45        """Holds column index alignments to treat alignments
 46        differently to the `css`. For example, to set the
 47         alignemnt of the first column to left, call 
 48         `set_column_alignments` with a parameter of `["left"]`"""

 52        Allow the default css to be overridden if necessary

 56
 57        self.css = "\n" + css + "\n"

197    def set_column_headings(self, headings: list = None) -> None:
198        """
199        Set columns to be headings (bold / different background) for each
200        column where the list element is True
201
202        :param headings: A list of boolean values marking a column as a
203            heading.  By default, the first column is set
204        :return: None
205        """
206        if not headings:
207            headings = [True]
208
209        self.column_headings = headings
210
211    def set_column_alignments(self, alignments: list = None) -> None:
212        """
213        Set column alignments to the values in `alignments`
214
215        :param alignments: A list of str values marking the justification of a
216        column: "left", "center" or "right".  If a list element is not set
217        then the `css` rules are followed. By default, the first column is
218        set to left justified
219        :return: None
220        """
221        if not alignments:
222            alignments = ["left"]
223
224        self.column_alignments = alignments
225
226    def start_row(self, style: str = "") -> None:
227        """
228        Start a new row in the current table
229
230        :param style: The css style to use.
231        :return: None
232        """
233        if not style:
234            self.body += "  <tr>\n"
235        else:
236            self.body += "  <tr style='%s'>\n" % style
237
238    def add_cell(self, value: Any, tag: str = 'd', style: str = "") -> None:
239        """
240        Add a single cell to the current table
241
242        :param value: This can be of any type, it will be converted to a string
243        :param tag: Can be set to 'h' to create a heading
244        :param style: A css style for the cell.  Do not include the `style` tags
245        """
246        if (self._current_column < len(self.column_headings) and
247                self.column_headings[self._current_column]):
248            # This column defined as a heading so mark as `th`
249            tag = 'h'
250        if (self._current_column < len(self.column_alignments) and
251                self.column_alignments[self._current_column]):
252            # This column has an alignment set, mark as so...
253            style += ("text-align: %s; " %
254                      self.column_alignments[self._current_column].lower())
255
256        if not style:
257            self.body += "    <t%s>%s</t%s>\n" % (tag, str(value), tag)
258        else:
259            self.body += "    <t%s style='%s'>%s</t%s>\n" % (tag, style, str(value), tag)
260        self._current_column += 1
261
262    def add_cells(self, values: list, tag: str = 'd', style: str = "") -> None:
263        """
264        Add a list of cells, possibly to a partial row
265        :param values: A list of values to add.  Elements can be of any type
266        :param tag: The tag: either 'd' or 'h'. Default is 'd'
267        :param style: A css style for the cells
268        :return: None
269        """
270        for v in values:
271            self.add_cell(v, tag, style)
272
273    def end_row(self) -> None:
274        """
275        End the current row in the current table
276        """
277        self.body += "  </tr>\n"
278        self._current_row += 1
279        self._current_column = 0
280
281    def add_row(self, values: list, tag: str = "d", style: str = ""):
282        """
283        Add a whole row to a table.  The values passed in should be a
284        list of values that make up the complete row
285
286        :param values: A list of values
287        :param tag: The tag for the row - 'h' or 'd'
288        :param style: a css style for the row (do not include `style` tags)
289        """
290
291        self.start_row(style)
292        for value in values:
293            self.add_cell(value, tag, style)
294            # self.body += "    <t%s>" % tag + str(td) + "</t%s>\n" % tag
295        self.end_row()
296
297    def add_headings(self, values: list) -> None:
298        """
299        Calls `add_row` with a tag of `h` and passes `values` to
300        :param values: a list of values to be used as headings
301        """
302        self.add_row(values, "h")
303
304    def end_table(self) -> None:
305        """
306        End the current table with a `/table` tag.  Not ending the table can
307        lead to unexpected results!
308        """
309        self.body += "</table>\n"
310
311    def add_html(self, value: str) -> None:
312        """
313        Add raw HTML to the class
314
315        :param value: The raw HTML
316        """
317        self.body += value + "\n"
318
319    def add_url(self, target: str, text: str = None) -> None:
320        """
321        Add a url to the object that will be displayed as a clickable link
322        :param target: The target address or URL
323        :param text: The text to display for the link. If ommitted, the target
324            name is used
325        """
326        if not text:
327            text = target
328        self.body += "<a href='%s'>%s</a>\n" % (target, text)
329
330    def get_html(self) -> str:
331        """
332        Put all the elements together and return a formatted HTML document
333
334        :return: The complete HTML document
335        """
336        value = self.doctype
337        value += self.html
338        value += self.head
339        value += "<style>" + self.css + "\n</style>\n"
340        value += "</head>\n"
341        value += self.body
342        value += "</body>\n"
343        value += "</html>\n"
344
345        return value
