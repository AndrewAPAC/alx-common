 20class Paths:
 21    def __init__(self, appname: str, inifile: str = None) -> None:
 22        """
 23        A class to hold the default paths for the application. These values are
 24        all determined from the installation location of the calling script.
 25
 26        If the `data` is needed, it must be created manually to avoid unnecessary
 27        data directories.
 28
 29        :param appname: The application name
 30        :param inifile: If the inifile name is different from the default `app.ini` then it can be set here
 31        """
 32
 33        basename = os.path.basename(os.path.dirname(sys.argv[0]))
 34        dirname = os.path.dirname(sys.argv[0])
 35        self.root = os.path.abspath(os.path.join(dirname, "..", ".."))
 36        """The root of the installation, perhaps `/opt/local/env` 
 37        (where *env* is *prod*, *test* or *dev*). Paths are calculated
 38         dynamically, based on file location"""
 39        self.bin = os.path.join(self.root, "bin")
 40        """The location of the start script: `root/bin`"""
 41        self.data = os.path.join(self.root, "data", basename)
 42        """The location of the application data: `root/data/app`"""
 43        self.etc = os.path.join(self.root, 'etc')
 44        """The location of the configuration files: `root/etc`"""
 45        self.log = os.path.join(self.root, "log", basename)
 46        """The location of the log files: `root/log/app`"""
 47        self.logfile = os.path.join(self.log, appname + ".log")
 48        """The log filename: `root/log/app/app.log`"""
 49        self.scripts = os.path.join(self.root, "scripts")
 50        """The location of the scripts: `root/scripts`"""
 51        self.top = os.path.join(self.root, "scripts", basename)
 52        """The location of the application scripts: `root/scripts/app`"""
 53        self.module_config_dir = self.get_module_config_dir()
 54        """The location of the `keyfile`, `local_env` and `local_config` files"""
 55        self.keyfile = os.path.join(self.module_config_dir, "key")
 56        """The file from where to obtain the key: `~/.config/alx/key` or
 57        `%APPDATA%\\alx\\key` on Windows. This file holds the `cryptography`
 58        key for use with `ALXapp.decrypt` and `ALXapp.encrypt`"""
 59        self.local_env = os.path.join(self.module_config_dir, "env")
 60        """The path to the environment to be used in use: `~/.config/alx/env`.
 61        Typically it only holds the python venv to be used:
 62        - `venv=/path/to/venv/current`"""
 63        self.local_config = os.path.join(self.module_config_dir, "alx.ini")
 64        """The path to the local configuration: `~/.config/alx/alx.ini`. Entries
 65        in this file will override those of the global configuration"""
 66        self.global_config = os.path.join(
 67            os.path.dirname(os.path.abspath(__file__)), "alx.ini")
 68        """The path to the module configuration: `<venv>/lib/.../site-packages/alx/alx.ini`"""
 69
 70        self.config = os.path.join(self.etc, appname + ".ini")
 71        """The name of the config file determined from `self.etc/app.ini`"""
 72        if inifile:
 73            self.config = os.path.join(self.etc, inifile)

 82    def create_configuration_files(self) -> None:
 83        """
 84        Create some initial configuration files under `module_config_dir`.
 85        Descriptions of the files are above:
 86        - `local_env`
 87        - `local_config`
 88        - `keyfile`
 89
 90        :return: None
 91        """
 92        os.makedirs(self.module_config_dir, exist_ok=True)
 93
 94        if not os.path.isfile(self.local_config):
 95            print("*** NOTE: Creating user config file '{}'".format(self.local_config))
 96            gc = ALXapp.read_config(self.global_config)
 97            with open(self.local_config, 'w') as f:
 98                for s in gc.sections():
 99                    f.write("[%s]\n\n" % s)
100
101        if not os.path.isfile(self.keyfile):
102            print("*** NOTE: Creating key file '{}'".format(self.keyfile))
103            # Create an initial fernet key.  User to adjust as required
104            with open(self.keyfile, 'w') as f:
105                f.write("%s\n" % Fernet.generate_key().decode())
106            # Ensure strict permissions
107            os.chmod(self.keyfile, 0o600)
108
109        if not os.path.isfile(self.local_env):
110            print("*** NOTE: Creating local environment file: '{}'".format(self.local_env))
111            # Assume that the current python invocation is the path to the virtual env
112            with open(self.local_env, 'w') as f:
113                venv = os.path.dirname(os.path.dirname(sys.executable))
114                f.write("venv=%s\n" % venv)

123    def __init__(self, description: str = "Unknown App",
124                 args: list = None, appname: str = None,
125                 inifile: str = None, epilog: str = None):
126        """
127        Initialise the `ALXapp` object which does a number of things:
128        * Creates the application name from `sys.argv[0]` and stores it in
129        `name` if not passed as a parameter
130        * Adds a `--env` / `-e` argument and sets `self.environment`
131        * Initialises the `Paths` class.
132        * Reads the arguments provided in the `args` parameter.
133        * Reads and parses the configuration file and stores the values in the ALXAppp object
134        * Reads and parses the library configuration stored in `alx.ini`
135        * Initialises and starts logging to `Paths.logfile`
136
137        An example `alx.ini` or `$HOME/config/alx/alx.ini` or
138        `%APPDATA%\alx` on Windows
139        ```
140        [DEFAULT]
141
142        [logging]
143        format:     %%(asctime)s: %%(levelname)s %%(module)s.%%(funcName)s:%%(lineno)d [%%(threadName)s] %%(message)s
144        loglevel:   INFO
145        days:       7
146        maxsize:    10485760
147        when:       midnight
148
149        [mail]
150        server:     mailhost
151        from:       Super User <root@localhost>
152
153
154        [html]
155        css:
156                p, h1, h2, h3, h4, h5, ul, ol, li {
157                    font-family: Calibri, Arial, Helvetica
158                }
159                table {
160                  font-family: "Trebuchet MS", Arial, Helvetica, sans-serif;
161                  border-collapse: collapse;
162                  width: 100%%;
163                }
164                td, th {
165                  border: 1px solid #ddd;
166                  padding: 5px;
167                }
168                tr:nth-child(even){background-color: #f2f2f2;}
169                tr:hover {background-color: #ddd;}
170                th {
171                  padding-top: 8px;
172                  padding-bottom: 8px;
173                  text-align: left;
174                  background-color: #0771af;
175                  color: white;
176                }
177        ```
178        :param description: A short description for the app - used with `--help`
179        :param args: A list of arguments added with argparse.ArgumentParser.add_argument.  Example
180        ```
181        args = [
182            ["-d", "--date", {"help": "store as '%%Y-%%m-%%d' date in database"}],
183            ["-s", "--start", {"default": None, "type": str,
184                               "help": "The first date from which to retrieve prices"}],
185            ["-g", "--gui", {"action": "store_true", "default": False,
186                             "help": "Display the browser."}],
187            ["locations", {nargs='*', default=[],
188                           "help": "List of locations"}]
189        ]
190        ```
191        but you can just use this trimmed down version to store the
192        arguments as strings
193        ```
194        args = [
195            ["--date"],
196            ["--start"],
197         ]
198        ```
199        :param appname: The name of the application.  Default is the
200         name of argv[0]
201        :param inifile: The name of the configuration file.  Default
202         is to create it from appname
203        :param epilog: The text to use at the end of the help
204         message when the app is called with `--help`
205        """
206
207        self.name = os.path.splitext(os.path.basename(sys.argv[0]))[0]
208        """The name of the application from the parameter or calculated from `sys.argv`"""
209        if appname:
210            self.name = appname
211
212        parser = argparse.ArgumentParser(description=description,
213                                         epilog=epilog,
214                                         formatter_class=argparse.ArgumentDefaultsHelpFormatter)
215
216        parser.add_argument("-e", "--env", default='dev',
217                            help="Runtime Environment, dev, test or prod")
218
219        if args:
220            for a in args:
221                if type(a[-1]) is dict:
222                    arg = a[:-1]
223                    kwargs = a[-1]
224                    parser.add_argument(*arg, **kwargs)
225                else:
226                    parser.add_argument(*a)
227
228        self.arguments = parser.parse_args()
229        """A namespace of the arguments as parsed by 
230        `argparse.Arguments.parse_args`"""
231        self.environment = 'dev'
232        """The execution environment - default is `dev` but any of 
233        * test || uat || tst: **test**
234        * prod || production || prd: **production**
235        
236        can be used to set *prod* or *test*
237        """
238        if self.arguments.env in ('test', 'uat', 'tst'):
239            self.environment = 'test'
240        elif self.arguments.env in ('prd', 'prod', 'production'):
241            self.environment = 'prod'
242
243        self.paths = Paths(self.name, inifile)
244        """The `Paths` namespace that holds path information"""
245
246        self.config = self.read_config(self.paths.config)
247        """The application configuration read from `Paths.config` from a 
248        call to `ALXApp.read_config`. The configuration values are assigned
249        to the `ALXApp` class with a call to `ALXApp.parse_config`"""
250        if self.config and self.environment in self.config:
251            self.parse_config(self, self.config[self.environment])
252
253        self.key = None
254        """The key to encrypt and decrypt encoded strings"""
255
256        self.libconfig = self.read_lib_config()
257        """The global library configuration from `alx.ini`"""
258
259        # Start logging with all configured parameters
260        self.start_logging()

263    def parse_config(obj: object, config: configparser.SectionProxy) -> object:
264        """
265        Parses the `config` and stores in `obj` as the appropriate type.
266        It works out if it is a boolean, float, integer or string. If
267        the configuration value starts with a `[` or `{` then it is
268        loaded using `json.loads` and stored in an `OrderedDict`
269
270        The config file should be structured so there are sections named
271        * `[DEFAULT]`
272        * `[dev]`
273        * `[test]`
274        * `[prod]`
275        This allows settings to be overridden simply by specifying
276        `-e env`. If a setting is not specified in an environment
277        section, then the value from `[DEFAULT]` is used. For example
278        ```
279        [DEFAULT]
280        loglevel:   INFO
281
282        [dev]
283        loglevel:   DEBUG
284
285        [test]
286        loglevel:   INFO
287
288        [prod]
289        loglevel:   WARN
290        ```
291        All the config file elements are stored in the `ALXapp` object.
292        If you define:
293        ```
294        foo: bar
295        bar: 1
296        baz: [1, 2, 3, 4]
297        ```
298        in the ini file, then your `ALXapp` object, say `app`, will have
299        these elements defined (either from the `[DEFAULT]` section or the
300        appropriate environment:
301        ```
302        app.foo = "bar"
303        app.bar = 1
304        app.baz = [1, 2, 3, 4]
305        ```
306        Additionally, if you specify `$data` in the value, it will be
307        expanded to the data directory of the application.  For example
308        ```
309        output_file = $data/output.txt
310        ```
311        will result in:
312        ```
313        app.output_file = /opt/local/prod/data/app/output.txt
314        ```
315
316        :param obj: The object in which to store the configuration
317         values (usually `self`)
318        :param config: The `configparser` object.  It can be a section
319        like `config[app.environment]`
320
321        :return: The object with typed configuration attributes added
322        """
323
324        try:
325            for item, value in config.items():
326                # Add all the config values as string values in the app
327                value = config.get(item)
328                if '$data' in value:
329                    if hasattr(obj, 'paths'):
330                        value = value.replace('$data', obj.paths.data)
331                        setattr(obj, item, value)
332                    else:
333                        continue
334                elif value in ('True', 'False', 'true', 'false'):
335                    # Convert to boolean
336                    setattr(obj, item, config.getboolean(item))
337                else:
338                    try:
339                        # Is it an integer?
340                        i = int(value)
341                        setattr(obj, item, i)
342                    except (ValueError, TypeError):
343                        try:
344                            # or a float?
345                            f = float(value)
346                            setattr(obj, item, f)
347                        except (ValueError, TypeError):
348                            # Only a string left....
349                            if value.startswith('[') or value.startswith('{'):
350                                try:
351                                    setattr(obj, item, json.loads(
352                                        value, object_pairs_hook=OrderedDict))  # type: ignore[arg-type]
353                                except json.JSONDecodeError:
354                                    setattr(obj, item, value)
355                            else:
356                                setattr(obj, item, value)
357        except Exception:
358            raise
359
360        return obj

362    def parse_config_section(self, obj, config, include_defaults=False) -> object:
363        """
364        Parse a config file section and store it in the object `obj`. This is
365        a wrapper for `parse_config` function to maintain compatibility
366        with `ALXapp.parse_config`.
367
368        :param obj: An object in which to store the configuration
369        :param config: The configparser object.  It can be a section
370         like `config[app.environment]`
371        :param include_defaults: Whether the values in the `[DEFAULT]` should
372        be parsed.
373
374        :return: The object with typed configuration attributes added
375        """
376
377        if include_defaults:
378            items = config.items()
379        else:
380            # Get keys that are explicitly defined in the section (not from defaults)
381            section_keys = set(self.config.options(config.name)) - set(self.config.defaults().keys())
382
383            items = {}
384            for key in config:
385                # Include only keys that are explicitly in this section
386                if key in section_keys:
387                    items[key] = config[key]
388                # Or keys in defaults that are overridden (have different values)
389                elif key in self.config.defaults() and config[key] != self.config.defaults()[key]:
390                    items[key] = config[key]
391                else:
392                    # First default-only key, stop
393                    break
394
395        # Reuse the existing parsing logic
396        return self.parse_config(obj, items)

419    def read_lib_config(self) -> configparser.ConfigParser:
420        """
421        Reads and parses the module configuration file and the local
422        configuration file (`~/.config/alx/alx.ini` or
423        `%APPDATA%/alx/alx.ini`). The local configuration overrides

442    def start_logging(self):
443        """
444        Sets up a standard logger using the configuration in `alx.ini`
445        By default, log files are rolled at midnight and kept for the
446        number of days configured in `alx.ini`.  The loglevel and
447        format are also set in the config file.
448
449        If the application is in development mode, a console logger
450        is also added for convenience
451        """
452        days = self.libconfig.getint('logging', 'days')
453        log_format = self.libconfig.get('logging', 'format')
454        when = self.libconfig.get('logging', 'when')
455
456        if hasattr(self, 'loglevel'):
457            loglevel = self.loglevel
458        else:
459            loglevel = self.libconfig.get('logging', 'loglevel')
460
461        self.logger.setLevel(loglevel)
462
463        if not os.path.isdir(self.paths.log):
464            os.makedirs(self.paths.log)
465
466        fh = TimedRotatingFileHandler(self.paths.logfile, when=when,
467                                      backupCount=days)
468        formatter = logging.Formatter(log_format)
469        fh.setLevel(loglevel)
470        fh.setFormatter(formatter)
471        self.logger.addHandler(fh)
472
473        if self.is_dev():
474            # Also log to console...
475            ch = logging.StreamHandler()
476            ch.setLevel(loglevel)
477            ch.setFormatter(formatter)
478            self.logger.addHandler(ch)
479
480        self.logger.debug("Starting application '%s', logging at level '%s'",
481                          format(self.name), format(loglevel))

496    def _read_key(self) -> Fernet:
497        """Reads the key from `~/.config/alx/key` and uses it to encrypt and
498         decrypt strings using the `cryptography` python module"""
499
500        if not os.path.exists(self.paths.keyfile):
501            self.logger.error("Could not open %s", self.keyfile)
502            sys.exit(1)
503        st = os.stat(self.paths.keyfile)
504        if platform.system() != 'Windows' and int(oct(st.st_mode)[3:]) > 600:
505            # Apologies to Windows users but your security is too messy
506            self.logger.error("Check permissions on %s.  Too open", self.keyfile)
507            sys.exit(1)
508
509        with open(self.paths.keyfile) as k:
510            key = k.read().strip()
511
512        fernet = Fernet(key)
513
514        return fernet

516    def encrypt(self, string: str) -> str:
517        """
518        Encrypts the password using the key read from `~/.config/alx/key`. This
519        function provides an easy way to avoid storing plain text passwords
520
521        The keyfile needs to have permission `0600` and look something like
522        ```
523        P9H7kabcdefghijwlSrfVnKKqV5VCnW5RE8OT21eH5k=
524        ```
525        Please refer to `decrypt` for how to create a key.
526
527        Note that due to the inconsistent implementation of permissions
528        on Windows, this check is ignored
529
530        :param string: The string to encrypt.
531        :return: The encrypted string
532        """
533        if not self.key:
534            self.key = self._read_key()
535        encoded = self.key.encrypt(string.encode())
536        return encoded.decode()
537
538    def decrypt(self, string: str) -> str:
539        """
540        Decrypts the password using the key read from `~/.config/alx/key`. This
541        function provides an easy way to avoid storing plain text passwords
542
543        Your code will then look like:
544        ```
545        self.passwd = app.decrypt(app.config.get('mysql', 'password'))
546        self.user = app.decrypt(app.config.get('mysql', 'user'))
547        ```
548        To generate a key, use this one-liner and place the ouput in `~/.config/alx/key`:
549        ```
550        python -c "from cryptography.fernet import Fernet; print(Fernet.generate_key().decode())"`
551        ```
552
553        If there are different users of an application using `ALXapp` then
554        the key must be consistent or the string will fail to decrypt.
555
556        :param string: The string to decrypt.
557        :return: The decrypted string
558        """
559        if not self.key:
560            self.key = self._read_key()
561        decoded = self.key.decrypt(string.encode()).decode()
562        return decoded

264    def parse_config(obj: object, config: configparser.SectionProxy) -> object:
265        """
266        Parses the `config` and stores in `obj` as the appropriate type.
267        It works out if it is a boolean, float, integer or string. If
268        the configuration value starts with a `[` or `{` then it is
269        loaded using `json.loads` and stored in an `OrderedDict`
270
271        The config file should be structured so there are sections named
272        * `[DEFAULT]`
273        * `[dev]`
274        * `[test]`
275        * `[prod]`
276        This allows settings to be overridden simply by specifying
277        `-e env`. If a setting is not specified in an environment
278        section, then the value from `[DEFAULT]` is used. For example
279        ```
280        [DEFAULT]
281        loglevel:   INFO
282
283        [dev]
284        loglevel:   DEBUG
285
286        [test]
287        loglevel:   INFO
288
289        [prod]
290        loglevel:   WARN
291        ```
292        All the config file elements are stored in the `ALXapp` object.
293        If you define:
294        ```
295        foo: bar
296        bar: 1
297        baz: [1, 2, 3, 4]
298        ```
299        in the ini file, then your `ALXapp` object, say `app`, will have
300        these elements defined (either from the `[DEFAULT]` section or the
301        appropriate environment:
302        ```
303        app.foo = "bar"
304        app.bar = 1
305        app.baz = [1, 2, 3, 4]
306        ```
307        Additionally, if you specify `$data` in the value, it will be
308        expanded to the data directory of the application.  For example
309        ```
310        output_file = $data/output.txt
311        ```
312        will result in:
313        ```
314        app.output_file = /opt/local/prod/data/app/output.txt
315        ```
316
317        :param obj: The object in which to store the configuration
318         values (usually `self`)
319        :param config: The `configparser` object.  It can be a section
320        like `config[app.environment]`
321
322        :return: The object with typed configuration attributes added
323        """
324
325        try:
326            for item, value in config.items():
327                # Add all the config values as string values in the app
328                value = config.get(item)
329                if '$data' in value:
330                    if hasattr(obj, 'paths'):
331                        value = value.replace('$data', obj.paths.data)
332                        setattr(obj, item, value)
333                    else:
334                        continue
335                elif value in ('True', 'False', 'true', 'false'):
336                    # Convert to boolean
337                    setattr(obj, item, config.getboolean(item))
338                else:
339                    try:
340                        # Is it an integer?
341                        i = int(value)
342                        setattr(obj, item, i)
343                    except (ValueError, TypeError):
344                        try:
345                            # or a float?
346                            f = float(value)
347                            setattr(obj, item, f)
348                        except (ValueError, TypeError):
349                            # Only a string left....
350                            if value.startswith('[') or value.startswith('{'):
351                                try:
352                                    setattr(obj, item, json.loads(
353                                        value, object_pairs_hook=OrderedDict))  # type: ignore[arg-type]
354                                except json.JSONDecodeError:
355                                    setattr(obj, item, value)
356                            else:
357                                setattr(obj, item, value)
358        except Exception:
359            raise
360
361        return obj
